fn main() {
    let students = vec!["Riya", "Manu", "Rose", "Anita"];

    let target = "Riya";
    let mut found = false;

    for name in &students {
        if *name == target {
            found = true;
            break;
        }
    }

    if found {
        println!("Student exists");
    } else {
        println!("Student missing");
    }
}
