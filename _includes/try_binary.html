fn main() {
    let mut nums = vec![40, 10, 70, 20, 90, 50];

    nums.sort();

    println!("Sorted: {:?}", nums);

    match nums.binary_search(&40) {
        Ok(index) => println!("Found at index {}", index),
        Err(_) => println!("Not found"),
    }
}
