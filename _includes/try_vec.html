fn main() {
  let mut students = Vec::new();

  //names are for illustration only//
  students.push("Riya");
  students.push("Anita");
  students.push("Manu");
  students.push("Rena");

  for s in &students {
    if s.starts_with('R'){
      println!("{s}");
    }
  }
}
