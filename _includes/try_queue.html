use std::collections::VecDeque;

fn main() {
    let mut queue = VecDeque::new();

    // push 4 values using a loop
    for i in 1..=4 {
        queue.push_back(i * 10);
    }

    // pop the first value
    queue.pop_front();

    //push one more value
    queue.push_back(50);

    println!("{:?}", queue);
}
