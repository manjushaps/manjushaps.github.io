use std::collections::BTreeMap;

fn main() {
    let mut marks = BTreeMap::new();

    marks.insert("Science", 95);
    marks.insert("Maths", 100);
    marks.insert("English", 90);
    marks.insert("Social", 80);
    marks.insert("Language", 88);

    for (subject, mark) in &marks {
        println!("{} -> {}", subject, mark);
    }
}
